title: Embeddings at Scale
---

Build embedding pipelines around a reviewable table contract: filter a deterministic candidate slice, keep stable IDs and source fields in the projection, add one embedding column, validate the materialized result, and only then write it. SQL Expression is the default path. The examples below start from prepared chunk rows so each vector already has a durable `(document_id, chunk_id)` key.

## Complete a Hosted Embedding Run with SQL

This first workflow uses a hosted provider and completes the task before introducing alternate APIs. `embedding_source` and `embedding_candidates` contain no provider output, while `embedding_model_output` materializes the external call once for validation and writing.

```python
import math

import vane

con = vane.connect()

source = con.sql(
    """
    SELECT
        CAST(document_id AS VARCHAR) AS document_id,
        CAST(chunk_id AS BIGINT) AS chunk_id,
        CAST(source_uri AS VARCHAR) AS source_uri,
        CAST(chunk_text AS VARCHAR) AS chunk_text
    FROM read_parquet('data/document-chunks/*.parquet')
    WHERE document_id IS NOT NULL
      AND chunk_id IS NOT NULL
      AND source_uri IS NOT NULL
      AND chunk_text IS NOT NULL
      AND length(trim(chunk_text)) > 0
    """
)
source.to_table("embedding_source")

candidates = con.sql(
    """
    SELECT
        'embedding-run-2026-07-11-001'::VARCHAR AS run_id,
        document_id,
        chunk_id,
        source_uri,
        chunk_text
    FROM embedding_source
    ORDER BY document_id, chunk_id
    LIMIT 1000
    """
)
candidates.to_table("embedding_candidates")

candidate_check = con.sql(
    """
    SELECT
        count(*) AS row_count,
        count(DISTINCT struct_pack(
            document_id := document_id,
            chunk_id := chunk_id
        )) AS distinct_id_count,
        count(*) FILTER (
            WHERE document_id IS NULL OR chunk_id IS NULL
        ) AS null_id_count,
        count(*) FILTER (
            WHERE source_uri IS NULL OR chunk_text IS NULL
        ) AS null_source_count
    FROM embedding_candidates
    """
).fetchone()
expected_row_count, distinct_id_count, null_id_count, null_source_count = candidate_check
assert expected_row_count > 0
assert distinct_id_count == expected_row_count
assert null_id_count == 0
assert null_source_count == 0

expected_ids = con.sql(
    """
    SELECT document_id, chunk_id
    FROM embedding_candidates
    ORDER BY document_id, chunk_id
    """
).fetchall()

embedded = con.sql(
    """
    SELECT
        run_id,
        document_id,
        chunk_id,
        source_uri,
        chunk_text,
        'openai'::VARCHAR AS embedding_provider,
        'text-embedding-3-small'::VARCHAR AS embedding_model,
        ai_embed(
            chunk_text,
            struct_pack(
                provider := 'openai',
                model := 'text-embedding-3-small',
                dimensions := 512,
                normalize := true,
                batch_size := 64,
                concurrency := 2,
                on_error := 'raise'
            )
        ) AS embedding
    FROM embedding_candidates
    ORDER BY document_id, chunk_id
    """
)
embedded.to_table("embedding_model_output")

output_check = con.sql(
    """
    SELECT
        count(*) AS row_count,
        count(DISTINCT struct_pack(
            document_id := document_id,
            chunk_id := chunk_id
        )) AS distinct_id_count,
        count(*) FILTER (
            WHERE document_id IS NULL OR chunk_id IS NULL
        ) AS null_id_count,
        count(*) FILTER (
            WHERE source_uri IS NULL OR chunk_text IS NULL
        ) AS null_source_count,
        count(*) FILTER (WHERE embedding IS NULL) AS null_embedding_count
    FROM embedding_model_output
    """
).fetchone()
(
    actual_row_count,
    actual_distinct_id_count,
    actual_null_id_count,
    actual_null_source_count,
    null_embedding_count,
) = output_check
assert actual_row_count == expected_row_count
assert actual_distinct_id_count == expected_row_count
assert actual_null_id_count == 0
assert actual_null_source_count == 0
assert null_embedding_count == 0

actual_ids = con.sql(
    """
    SELECT document_id, chunk_id
    FROM embedding_model_output
    ORDER BY document_id, chunk_id
    """
).fetchall()
assert actual_ids == expected_ids

result = con.sql(
    """
    SELECT
        run_id,
        document_id,
        chunk_id,
        source_uri,
        chunk_text,
        embedding_provider,
        embedding_model,
        embedding
    FROM embedding_model_output
    ORDER BY document_id, chunk_id
    """
)
expected_columns = [
    "run_id",
    "document_id",
    "chunk_id",
    "source_uri",
    "chunk_text",
    "embedding_provider",
    "embedding_model",
    "embedding",
]
expected_types = [
    "VARCHAR",
    "VARCHAR",
    "BIGINT",
    "VARCHAR",
    "VARCHAR",
    "VARCHAR",
    "VARCHAR",
    "FLOAT[512]",
]
assert list(result.columns) == expected_columns
assert [str(dtype) for dtype in result.types] == expected_types

for document_id, chunk_id, vector in con.sql(
    """
    SELECT document_id, chunk_id, embedding
    FROM embedding_model_output
    """
).fetchall():
    assert document_id is not None and chunk_id is not None
    assert len(vector) == 512
    assert all(math.isfinite(float(value)) for value in vector)
    norm = math.sqrt(sum(float(value) ** 2 for value in vector))
    assert math.isclose(norm, 1.0, rel_tol=1e-4, abs_tol=1e-4)

result.write_parquet("output/embedding-run-2026-07-11-001/part-00000.parquet")
```

The second argument to `ai_embed` is one foldable constant `STRUCT`: provider, model, vector shape, normalization, batching, actor count, and error policy bind once for the task rather than becoming columns or per-row options. SQL binding rejects credential-like fields. Put `OPENAI_API_KEY` and any endpoint-specific secrets in every worker environment, never in SQL text or serialized options.

The `LIMIT` is safe only with a deterministic `ORDER BY`. Increase it after the complete check passes; do not replace the stable business key with a generated row number.

## Choose Vector Shape and Normalization

Without `dimensions`, the declared SQL and Python Expression type is dynamic `FLOAT[]`. Set `dimensions=N` only when the selected provider and model support that size; the result is fixed `FLOAT[N]`, which makes schema drift and wrong-length vectors easier to reject before indexing.

`normalize=true` requests L2 normalization. Choose it at task level according to the retrieval metric and index contract, then verify finite values, exact length, and approximately unit norm as the completed workflow does. Do not infer correctness from a provider request succeeding. See the [AI Function API Reference](/docs/data/reference/ai-api) for exact provider, dimension, normalization, retry, and error options.

## Tune Batches, Concurrency, and Workers

- `batch_size` controls how many input rows reach an AI actor in one UDF batch. The OpenAI embedding implementation may further pack requests under its token budget. Increase batch size until memory, latency, or failure scope becomes the limiting factor.
- SQL `concurrency` maps to the AI actor count. It is not a per-row option and does not itself mean that one actor sends that many simultaneous requests. Request-level controls such as `max_api_concurrency` are provider- and capability-specific; do not assume every embedding provider consumes them.
- More actors can raise throughput, but also increase quota pressure, retries, cost, and duplicate external effects after task retries. Measure a bounded slice first.
- The active runner resolves SQL and Python Expression actors to local subprocesses or Ray actors. Local subprocesses inherit visible devices but do not reserve or isolate GPUs. Configure Ray before constructing the relation when scheduling, distributed scans, or GPU placement require it.
- Every worker needs the provider package, network and storage access, credential environment variables, and—for local models—the same model files or download access. Keep model and dependency versions identical across workers.

For vLLM GPU operation and actor-pool validation, continue with [GPU Inference](/docs/data/guides/gpu-inference).

## Use Python Expression as an Alternate

Use Python Expression when typed configuration belongs in Python. It has the same one-column, row-preserving projection semantics as the SQL workflow, including the same IDs and source fields.

```python
import vane

con = vane.connect()
candidates = con.sql(
    """
    SELECT
        'embedding-run-2026-07-11-001'::VARCHAR AS run_id,
        CAST(document_id AS VARCHAR) AS document_id,
        CAST(chunk_id AS BIGINT) AS chunk_id,
        CAST(source_uri AS VARCHAR) AS source_uri,
        CAST(chunk_text AS VARCHAR) AS chunk_text
    FROM read_parquet('data/document-chunks/*.parquet')
    WHERE document_id IS NOT NULL
      AND chunk_id IS NOT NULL
      AND source_uri IS NOT NULL
      AND chunk_text IS NOT NULL
      AND length(trim(chunk_text)) > 0
    ORDER BY document_id, chunk_id
    LIMIT 1000
    """
)

python_embedded = candidates.select(
    vane.col("run_id"),
    vane.col("document_id"),
    vane.col("chunk_id"),
    vane.col("source_uri"),
    vane.col("chunk_text"),
    vane.lit("openai").alias("embedding_provider"),
    vane.lit("text-embedding-3-small").alias("embedding_model"),
    vane.ai.embed(
        vane.col("chunk_text"),
        provider="openai",
        model="text-embedding-3-small",
        provider_options=vane.ai.OpenAIProviderOptions(
            concurrency=2,
        ),
        embedding_options=vane.ai.OpenAIEmbeddingOptions(
            on_error="raise",
        ),
        dimensions=512,
        normalize=True,
    ).alias("embedding"),
)
```

Apply the SQL workflow's row-count, ordered-key, null, type, dimension, finite-value, normalization, and run-specific write checks before publishing this alternate. Keep the AI call directly inside `rel.select(...)` with an alias; do not manually combine a result-only column with source rows by position.

## Run Local Transformers Locally or on Ray

For an offline or private path, rebuild the same Expression with a local Transformers model. The default runner uses a local subprocess actor. Configure Ray before creating the connection and relation when the execution must use Ray.

```python
import vane

vane.configure(runner="ray")
con = vane.connect()
candidates = con.sql(
    """
    SELECT
        CAST(document_id AS VARCHAR) AS document_id,
        CAST(chunk_id AS BIGINT) AS chunk_id,
        CAST(source_uri AS VARCHAR) AS source_uri,
        CAST(chunk_text AS VARCHAR) AS chunk_text
    FROM read_parquet('data/document-chunks/*.parquet')
    WHERE chunk_text IS NOT NULL
    ORDER BY document_id, chunk_id
    LIMIT 1000
    """
)

ray_embedded = candidates.select(
    vane.col("document_id"),
    vane.col("chunk_id"),
    vane.col("source_uri"),
    vane.col("chunk_text"),
    vane.ai.embed(
        vane.col("chunk_text"),
        provider="transformers",
        model="sentence-transformers/all-MiniLM-L6-v2",
        provider_options={"batch_size": 64},
        dimensions=384,
        normalize=True,
    ).alias("embedding"),
)
```

The model is instantiated lazily and reused across batches in its actor. The Transformers provider requests a GPU resource when CUDA is detected; Ray can reserve that resource, while a local subprocess cannot. Switching runners does not by itself promise more model actors, so validate the physical actor count, GPU assignment, model reuse, and output contract before scaling.

## Keep Outputs Reviewable and Downstream Writes Idempotent

Hosted provider calls are external effects outside SQL transactions and are not exactly-once. A retry can repeat a request and charge even if the original query never publishes a table. Preserve stable IDs, source text or a durable source reference, provider/model/config, and `run_id` beside every vector so inputs and outputs remain reviewable.

Write each provider run to a unique path. For a durable index or table, merge or replace by `(run_id, document_id, chunk_id)`—or by a versioned business key when a run ID should be superseded—instead of blindly appending retries. Keep candidate and materialized model-output tables separate so validation never reissues the provider call.

## Use Relation Embedding Only for Built-In Long-Text Chunking

The Python Relation helper is useful when its built-in `max_chunk_chars` and `chunk_overlap_chars` behavior is the capability the task needs, or when an explicit Relation backend and actor pool are required. Vane exposes no direct SQL Relation or table-function API for this helper.

```python
import vane

vane.configure(runner="ray")
con = vane.connect()
long_text_source = con.sql(
    """
    SELECT
        CAST(document_id AS VARCHAR) AS document_id,
        CAST(source_uri AS VARCHAR) AS source_uri,
        CAST(text AS VARCHAR) AS source_text
    FROM read_parquet('data/documents/*.parquet')
    WHERE text IS NOT NULL
    ORDER BY document_id
    LIMIT 100
    """
)

embedding_only = long_text_source.embed_text(
    "source_text",
    provider="openai",
    model="text-embedding-3-small",
    dimensions=512,
    output_column="embedding",
    max_chunk_chars=2000,
    chunk_overlap_chars=200,
    batch_size=64,
    actor_number=2,
    num_gpus=0,
    execution_backend="ray_actor",
)
```

Here `actor_number` is an explicit Relation UDF actor-pool control, distinct from configuring provider `concurrency` on an Expression; the actor backend must match it. The helper splits each input internally, combines chunk vectors with a text-length-weighted average, normalizes the combined vector, and declares its configured output column as dynamic `FLOAT[]` even when provider dimensions are requested.

`embedding_only` is deliberate: the helper does not promise to retain `document_id`, `source_uri`, or `source_text`, and it does not emit one output row per internal chunk. Never reconnect it to source rows by position. Use the SQL or Python Expression path when stable source columns must accompany each vector, and use the custom preprocessing stage when chunks need independent IDs and rows.

## Use a Custom Relation UDF Only for Table-Shaped Preprocessing

Use a custom Relation UDF when preprocessing must change cardinality or return several columns, such as turning documents into durable chunk rows. Do not replace the row-preserving embedding Expression with a custom UDF merely to add one vector column.

```python
import pyarrow as pa

import vane

con = vane.connect()
source = con.sql(
    """
    SELECT
        CAST(document_id AS VARCHAR) AS document_id,
        CAST(source_uri AS VARCHAR) AS source_uri,
        CAST(text AS VARCHAR) AS source_text
    FROM read_parquet('data/documents/*.parquet')
    WHERE document_id IS NOT NULL
      AND source_uri IS NOT NULL
      AND text IS NOT NULL
    """
)


def chunk_documents(batch: pa.Table) -> pa.Table:
    document_ids = batch.column("document_id").to_pylist()
    source_uris = batch.column("source_uri").to_pylist()
    source_texts = batch.column("source_text").to_pylist()
    out_document_id = []
    out_chunk_id = []
    out_source_uri = []
    out_chunk_text = []
    out_chunker_version = []

    for document_id, source_uri, source_text in zip(
        document_ids,
        source_uris,
        source_texts,
    ):
        parts = [
            part.strip()
            for part in str(source_text).split(".")
            if part.strip()
        ]
        for chunk_id, chunk_text in enumerate(parts):
            out_document_id.append(document_id)
            out_chunk_id.append(chunk_id)
            out_source_uri.append(source_uri)
            out_chunk_text.append(chunk_text)
            out_chunker_version.append("sentence-v1")

    return pa.table(
        {
            "document_id": pa.array(out_document_id, type=pa.string()),
            "chunk_id": pa.array(out_chunk_id, type=pa.int64()),
            "source_uri": pa.array(out_source_uri, type=pa.string()),
            "chunk_text": pa.array(out_chunk_text, type=pa.string()),
            "chunker_version": pa.array(
                out_chunker_version,
                type=pa.string(),
            ),
        }
    )


chunks = source.map_batches(
    chunk_documents,
    schema={
        "document_id": "VARCHAR",
        "chunk_id": "BIGINT",
        "source_uri": "VARCHAR",
        "chunk_text": "VARCHAR",
        "chunker_version": "VARCHAR",
    },
    batch_size=512,
    execution_backend="subprocess_task",
)
chunks.to_table("embedding_chunks")
```

The example is intentionally simple. Production chunkers should be tokenizer-aware, bound chunk size and overlap to the model contract, and version the algorithm. Validate unique `(document_id, chunk_id)` keys and deterministic output before sending `embedding_chunks` through the SQL workflow.
